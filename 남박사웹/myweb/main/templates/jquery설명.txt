<script>
    function checkForm(){
        if($.trim($("#email").val() == ""){  
            alert("이메일을 입력하세요");
            $("#email").focus();
            return false;
        }
        if($("#pass").val() == ""){
            alert("비밀번호를 입력하세요")
            $("#pass").focus();
            return false;    
        }
        return true;
    }
